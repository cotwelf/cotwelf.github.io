audio

比较详细的用法戳这里的链接，以下只记录了我用到的一些方法，可能不够全面。

// Phaser.Scene
import Phaser from 'phaser'

export default class MyScene extends Phaser.Scene {
  constructor(){
    super('myScene')
  }
  private myAudio
  preload(){
    // 加载
    this.load.audio('key', '{url of audio source}')
  }
  create() {
    this.myAudio = this.sound.add('key', {
      // config
      volume: 0.3, // 音量：0~1, 默认是 1
      loop: true, // 是否循环播放：boolean
    })
    this.myAudio.play({
      volume: 0.5
    })
    // 相当于以下写法，但它只可以设置一次，之后无法控制，建议还是赋值一个变量
    // this.sound.play(key, {
    //   volume: 0.5,
    //   loop: true,
    // });
  }
  private handleMyAudio = () => {
    this.myAudio.play({
      volume: 0.5 // 这里 config 会覆盖初始化 sound 的 config
    })
    this.myAudio.pause()
    this.myAudio.stop()
    this.myAudio.setVolume(0.2) // 设置 volume
    let volume = this.myAudio.volume // 获取 volume
    // 聪明的你应该可以知道如何修改其他 config 了
    var isPlaying = this.myAudio.isPlaying
    var isPaused = this.myAudio.isPaused
  }
}

那么问题来了，如果我有很多个 Scene，sound 会不会不间断捏？

答案是：会的

那么问题又来了：如果我在 SceneA 中定义了一个变量，赋值了个 audio 对象，如何在 SceneB 中操作它呢？（甚至 SceneC、SceneD）

我这里采取了比较笨的方法，不在单个 Scene 中存储变量，而是存到 cache 里

cache

cache 给我一种粗浅的理解，是一个帮你分好类的 localstorage，比如你可以直接存储数据到 this.cache.audio 中，而且可以是对象等引用类型（只是 cache 并不会帮你校验或者限制，比如你完全可以在 this.cache.audio 中存储一个布尔值或者 string）。举个栗子：

import click from '{PATH TO click.mp3}'
import bgmDark from '{PATH TO bgm-dark.mp3}'

export const ASSET_KEYS = {
  AUDIO: {
    CLICK: {
      KEY: 'click',
      SOURCE: click,
    },
    BGM_DARK: {
      KEY: 'bgm-dark',
      SOURCE: bgmDark,
    },
  }
}

export const handleAssets = {
  load: function(that){
    Object.keys(ASSET_KEYS.AUDIO).forEach((sound) => {
      that.load.audio(`${ASSET_KEYS.AUDIO[sound].KEY}-temp`, ASSET_KEYS.AUDIO[sound].SOURCE)
    })
  },
  add: function(that) {
    Object.keys(ASSET_KEYS.AUDIO).forEach((sound) => {
      that.cache.audio.add(
        `${ASSET_KEYS.AUDIO[sound].KEY}`,
        that.sound.add(`${ASSET_KEYS.AUDIO[sound].KEY}-temp`,
      ))
      that.cache.audio.remove(`${ASSET_KEYS.AUDIO[sound].KEY}-temp`)
    })
  },
  play: function(that: Phaser.Scene, key, config?: Phaser.Types.Sound.SoundConfig) {
    that.cache.audio.get(key).play(config)
  },
  stop: function(that, key) {
    that.cache.audio.get(key).stop()
  }
}

// 使用：在最初一个 Scene 里的 preload 中调用 handleAssets.load(this) 用以载入数据，create 中调用 handleAssets.add(this)，接下来就可以在任意一个 Scene 使用 handleAssets.play(this, 'someKey') 等

其实也可以不要 that，比如最后的 stop，可以写成 function(key) { ... }，调用方法 handleAssets.stop.bind('someKey')
