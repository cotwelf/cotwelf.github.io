假设我们有个需求，页面刚进入的时候放个定时器，1s 后 count + 1。我是这么干的

import React, { useEffect, useState } from 'react'
import * as ReactDOM from 'react-dom'

const DiyComponent = () => {
  const [count, setCount] = useState(0)
  const printCount = () => {
    setCount(count + 1)
  }
  useEffect(() => {
    const counter = setTimeout(() => {
      printCount()
    }, 1000)
    return () => {
      clearTimeout(counter)
    }
  }, [])
  return (
    <div>{count}</div>
  )
}

ReactDOM.render(<DiyComponent />, document.getElementsByClassName('container')[0])
