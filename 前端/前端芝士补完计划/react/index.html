1. React 中使用 useEffect 如何清除副作用

在 useEffect 中返回一个函数

// 例如我们在组件销毁时清除 setTimeout
const Component = () => {
  const [name, setName] = useState('')
  let timer = null
  useEffect(() => {
    timer = setTimeout(() => {
      setName('cotwelf')
    }, 1000)
    return () => {
      clearTimeout(timer)
      timer = null
      console.log('cleared')
    }
  }, [])
  return <div>{name}</div>
}
