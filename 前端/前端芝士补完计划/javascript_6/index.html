怎样判断一个变量是否是对象，在我看来，可以 new 出来的都是对象

函数声明与函数表达式的区别

fn1() // 2233
fn2() // Uncaught TypeError: fn2 is not a function

function fn1 () {
  console.log('fn1') // fn1
  fn3() //fn3
}
var fn2 = function () {
  console.log('fn2')
}
function fn3() {
  console.log('fn3')
}

在代码开始执行之前，解析器就已经通过一个名为 函数声明提升（function declaration hoisting） 的过程，读取并将函数声明添加到执行环境中。对代码求值时，JS 引擎在第一遍会声明函数并将它们放到源代码树的顶部，变量提升结束之后才会开始执行。执行到函数所在的语句之前，变量 fn2 中不会保存有对函数的引用。

apply() 和 call()

都是在特定的作用域中调用函数，能够扩充函数赖以运行的作用域，好处是对象不需要与方法有任何耦合关系。

window.color = "red"
var o = { color: "blue" }
function sayColor() {
  console.log(this.color)
}
sayColor(); // red
sayColor.call(this) // red
sayColor.call(window) // red
sayColor.call(o) // blue

// bind() 方法会创建一个函数的实例，低版本浏览器不支持
var objectSayColor = sayColor.bind(o)
objectSayColor() // blue

函数绑定

创建一个函数，可以在特定的 this 环境中以指定参数调用另一个函数

var handler = {
  message: "Event handled",

  handleClick: function(event){
    alert(this.message + ":" + event.type);
  }
};
var btn = document.getElementById("my-btn");
btn.addEventListener("click", handle.handleClick) // 这样调用 handleClick 的话，this 指向的是 btn，而不是 handle，所以 handle.handleClick 会是 undefined
// 方法一：闭包
btn.addEventListener("click", function(event) {
  handle.handleClick(event)
})
btn.addEventListener("click", handle.handleClick.bind(handler)) // 这样调用 handleClick 的话，this 指向的是 btn，而不是 handle

函数柯里化

编写一个 add() 函数，能正常执行下面的代码，并能在控制台输出注释中的数字

var add = function(){
  let args = Array.prototype.slice.call(arguments)
  let inner = function() {
    args.push(...arguments)
    return inner
  }
  // inner.toString = function () {
  //   return args.reduce(function(prev, cur) {
  //     return prev + cur
  //   })
  // }
  return inner
}
console.log(add(1, 2)) // 3
console.log(add(1, 2, 3)) // 6
console.log(add(1)(2)) // 3

柯里化（currying）也叫不完全函数，是一种部分求值的技术，能把一个完整的函数调用分解成多次函数调用，每次只传入部分函数，返回一个接受剩下参数的函数，如此循环往复，直将所有参数传递过去，最后得出结果。
